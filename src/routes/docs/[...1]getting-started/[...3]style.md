<script lang="ts">
	// import { SocialLink } from '@svelteness/kit-docs';
	import { Formly, type IField } from 'svelte-formly-light';

	const form_name = 'myform';
	const fields: IField[] = [
		{
			type: 'input',
			name: 'number_a',
			attributes: {
				type: 'number',
				id: 'number_a',
				label: 'Number A',
				placeholder: 'Enter number A'
			},
			rules: ['required']
		},
		{
			type: 'input',
			name: 'number_b',
			attributes: {
				type: 'number',
				id: 'number_b',
				label: 'Number B',
				placeholder: 'Enter number B'
			},
			rules: ['required']
		},
		{
			type: 'input',
			name: 'total',
			attributes: {
				id: 'total',
				type: 'number',
				label: 'Total = Number A + Number B'
			},
			preprocess: (field: IField, fields: IField[], values: any) => {
				if (values.touched === 'number_a' || values.touched === 'number_b') {
					field.value = parseInt(values.number_a) + parseInt(values.number_b);
				}
				return field;
			}
		}
	];

	let data = {};
	const onSubmit = ({ detail }: any) => {
		console.log('detail onSubmit', detail);
		data = detail;
	};

	const onUpdate = ({ detail }: any) => {
		console.log('detail onUpdate', detail);
		data = detail;
	};
</script>

<Formly
	{fields}
	{form_name}
	on:submit={onSubmit}
	on:update={onUpdate}
	btnSubmit={{ text: 'Submit2' }}
	realtime={true}
/>
